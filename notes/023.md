# Merge k Sorted Lists


## 题目

Merge k sorted linked lists and return it as one sorted list. Analyze and describe its complexity.

## 代码

```cpp
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        int sz = lists.size();
        ListNode* head = new ListNode(0);
        ListNode* last = head;
        int count = 0;
        for(int i=0; i<sz; ++i){
            ListNode* temp = lists[i];
            while(temp){
                count++;
                ListNode* node = new ListNode(temp->val);
                last->next = node;
                last = node;
                temp = temp->next;
            }
        }
        ListNode* free_ptr = head;
        head = head->next;
        delete(free_ptr);
        if(count!=0)
            head = merge_K_sort(head,count);
    
        return head;
    }
    
    ListNode * merge_K_sort(ListNode *head,int count)
    {
        if(count!=0 && count!=1){
            ListNode* cur = head;
            ListNode* forehead = cur;
            ListNode* key_ptr = cur;
            
            int key = cur->val;
            int front_count=0,back_count=0;
            cur = cur->next;
            while(cur && --count){
                if(cur->val > key){
                    forehead=cur;
                    back_count++;//后部分结点个数计数
                }
                else{
                    front_count++;//后部分几点个数计数
                    //将该结点连接到最前面
                    forehead->next = cur->next;
                    cur->next = head;
                    head = cur;//更新head
                    cur = forehead;
            
                }
                cur = cur->next;
            }//while
            head = merge_K_sort(head,front_count);
            key_ptr->next = merge_K_sort(key_ptr->next,back_count);
            return head;
            
        }
        return head;   
    }
    
    
};
```
